#include <LiquidCrystal.h>
LiquidCrystal lcd(2,3,4,5,6,7);
int cont=0;
bool clock=true;
int arr[8];
int inicial=3;
int times,bandera,inx,repeat;
void setup()
{
  pinMode(13, INPUT);
  pinMode(12, INPUT);
  Serial.begin(9600);
  pinMode(10,INPUT);
  lcd.begin(16,2);
  
}

void loop()
{
  cont=0;
  while(cont!=8){
  if(digitalRead(12)==1 & clock==true){
  Serial.print(digitalRead(13));
  arr[cont]=digitalRead(13);
  clock=false;
  cont++;
  }
  else if(digitalRead(12)==0){
   clock=true;
  }
  }
  Serial.println();

 int salida=0;
 int potencia=1;
 int c=7;
 for(int i=0;i<8;i++){
   for(int j=1;j<=c;j++){
     potencia=potencia*2;
   }
   //Serial.println(potencia);
   salida=salida+(potencia*arr[i]);
   potencia=1;
   c--;
 }
 Serial.println(salida);
  if(inicial>0){
    if(inicial==3){
      times=salida;
      Serial.println("times");
    }
    if(inicial==2){
      bandera=salida;
      Serial.println("bandera");
    }
    if(inicial==1){
      inx=salida;
      Serial.println("inx");
    }
    inicial--;
  }
  else if(inx!=0){
    if(digitalRead(10)==0){
      repeat++;
    }
    if(repeat==times){
      Serial.println(char(salida));
      repeat=0;
      lcd.print(char(salida));
    }
    inx--;
  }
}
